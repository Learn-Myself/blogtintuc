# User manual simplecrud-nestjs
    Install Docker
    https://docs.docker.com/docker-for-windows/install/

# Step 1: NPM Installation
    $ npm install"

# Step 2: Docker-compose
    # Bring up the docker with database
    $ docker-compose up
    
# Step 3: Open another CMD & Running the app
    # Development mode
    $ npm run start
    
    # Watch mode
    $ npm run start:dev
    
    # Production mode
    $ npm run start:prod
    
# Access website URL
    http://localhost:3000/
